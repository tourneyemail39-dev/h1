import java.security.InvalidKeyException;
import java.security.NoSuchAlgorithmException;
import java.util.Base64;
import javax.crypto.BadPaddingException;
import javax.crypto.Cipher;
import javax.crypto.IllegalBlockSizeException;
import javax.crypto.KeyGenerator;
import javax.crypto.NoSuchPaddingException;
import javax.crypto.SecretKey;

public class DES {
    public static void main(String[] args) throws NoSuchAlgorithmException,
            NoSuchPaddingException, InvalidKeyException,
            IllegalBlockSizeException, BadPaddingException {

        Base64.Encoder encoder = Base64.getEncoder();

        // Generate DES key
        KeyGenerator mygenerator = KeyGenerator.getInstance("DES");
        SecretKey myDesKey = mygenerator.generateKey();

        // Create Cipher object
        Cipher desCipher = Cipher.getInstance("DES");

        // Encryption
        desCipher.init(Cipher.ENCRYPT_MODE, myDesKey);
        byte[] myBytes = "This message is important".getBytes();
        byte[] myEncryptedBytes = desCipher.doFinal(myBytes);

        // Decryption
        desCipher.init(Cipher.DECRYPT_MODE, myDesKey);
        byte[] myDecryptedBytes = desCipher.doFinal(myEncryptedBytes);

        // Output
        System.out.println("Encrypted text: " + encoder.encodeToString(myEncryptedBytes));
        System.out.println("Decrypted text: " + new String(myDecryptedBytes));
    }
}

import java.security.MessageDigest;

public class MD5 {
    public static String getHash(byte[] inputBytes, String algorithm) {
        StringBuilder hashValue = new StringBuilder();
        try {
            // Create MessageDigest instance for given algorithm
            MessageDigest messageDigest = MessageDigest.getInstance(algorithm);
            messageDigest.update(inputBytes);

            // Generate the digest (hash)
            byte[] digestedBytes = messageDigest.digest();

            // Convert bytes to hex
            for (byte b : digestedBytes) {
                hashValue.append(String.format("%02x", b & 0xff));
            }
        } catch (Exception e) {
            System.out.println("Error: " + e);
        }
        return hashValue.toString();
    }

    public static void main(String[] args) {
        String someString = "this is some string";

        // Compute MD5 hash
        String md5Hash = getHash(someString.getBytes(), "MD5");

        System.out.println("Input String: " + someString);
        System.out.println("MD5 Digest : " + md5Hash);
    }
}

import java.util.*;

class MonoalphabeticCipher {
    public String encode(String ptxt, String key) {
        String ciphertext = "";
        for (int i = 0; i < ptxt.length(); i++) {
            int index = (int) ptxt.charAt(i) - 65;
            char ctxt = key.charAt(index);
            ciphertext += ctxt;
        }
        return ciphertext;
    }

    public String decode(String ctxt, String key) {
        String plaintext = "";
        for (int i = 0; i < ctxt.length(); i++) {
            char ct = ctxt.charAt(i);
            int index = key.indexOf(ct);
            char pt = (char) (index + 65);
            plaintext += pt;
        }
        return plaintext;
    }
}

public class Monoalphabetic {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        String key = "QWERTYUIOPASDFGHJKLZXCVBNM";

        System.out.print("Enter the plain text: ");
        String ptxt = sc.next();

        MonoalphabeticCipher mono = new MonoalphabeticCipher();
        String ctxt = mono.encode(ptxt.toUpperCase(), key);
        System.out.println(ctxt);

        String plaintxt = mono.decode(ctxt, key);
        System.out.println(plaintxt);
    }
}

import java.util.*;

class Rail {
    public void encode(String pt, int key) {
        char[][] mat = new char[key][pt.length()];
        int row = 0;
        boolean down = true;

        // Initialize with '_'
        for (int i = 0; i < key; i++) {
            Arrays.fill(mat[i], '_');
        }

        // Fill matrix in zigzag
        for (int col = 0; col < pt.length(); col++) {
            mat[row][col] = pt.charAt(col);
            if (down) {
                row++;
                if (row == key) {
                    row = key - 2;
                    down = false;
                }
            } else {
                row--;
                if (row < 0) {
                    row = 1;
                    down = true;
                }
            }
        }

        // Print the matrix
        System.out.println("\nRail Fence Matrix:");
        for (int i = 0; i < key; i++) {
            for (int j = 0; j < pt.length(); j++) {
                System.out.print(mat[i][j] + " ");
            }
            System.out.println();
        }

        // Read row-wise for ciphertext
        StringBuilder ctext = new StringBuilder();
        for (int i = 0; i < key; i++) {
            for (int j = 0; j < pt.length(); j++) {
                if (mat[i][j] != '_') {
                    ctext.append(mat[i][j]);
                }
            }
        }

        System.out.println("\nCipher Text: " + ctext.toString());
    }
}

public class RailFenceCipher {

import java.util.*;

class CeaserCipher {
    public String encode(String ptext, int key) {
        String ciphertext = "";
        for (int i = 0; i < ptext.length(); i++) {
            int index = (int) ptext.charAt(i);
            char ch = (char) ((index + key - 65) % 26 + 65);
            ciphertext += ch;
        }
        return ciphertext;
    }

    public String decode(String ctext, int key) {
        String plaintext = "";
        for (int i = 0; i < ctext.length(); i++) {
            int index = (int) ctext.charAt(i);
            char ch = (char) ((index - key - 65 + 26) % 26 + 65);
            plaintext += ch;
        }
        return plaintext;
    }
}

#aes

import java.security.InvalidKeyException;
import java.security.NoSuchAlgorithmException;
import java.util.Base64;
import javax.crypto.BadPaddingException;
import javax.crypto.Cipher;
import javax.crypto.IllegalBlockSizeException;
import javax.crypto.KeyGenerator;
import javax.crypto.NoSuchPaddingException;
import javax.crypto.SecretKey;

public class AES {
    static Cipher cipher;

    // Encrypt function
    public static String encrypt(String ptext, SecretKey secretKey)
            throws InvalidKeyException, IllegalBlockSizeException, BadPaddingException {
        byte[] plaintextBytes = ptext.getBytes();
        cipher.init(Cipher.ENCRYPT_MODE, secretKey);
        byte[] encryptedBytes = cipher.doFinal(plaintextBytes);
        Base64.Encoder encoder = Base64.getEncoder();
        return encoder.encodeToString(encryptedBytes);
    }

    // Decrypt function
    public static String decrypt(String encrypted, SecretKey secretKey)
            throws InvalidKeyException, IllegalBlockSizeException, BadPaddingException {
        Base64.Decoder decoder = Base64.getDecoder();
        byte[] encryptedBytes = decoder.decode(encrypted);
        cipher.init(Cipher.DECRYPT_MODE, secretKey);
        byte[] decryptedBytes = cipher.doFinal(encryptedBytes);
        return new String(decryptedBytes);
    }

    public static void main(String[] args) throws NoSuchAlgorithmException,
            NoSuchPaddingException, InvalidKeyException,
            IllegalBlockSizeException, BadPaddingException {

        // Generate AES key
        KeyGenerator keyGenerator = KeyGenerator.getInstance("AES");
        keyGenerator.init(128); // AES-128
        SecretKey secretKey = keyGenerator.generateKey();

        cipher = Cipher.getInstance("AES");

        String plaintext = "This message is important";

        System.out.println("Before Encryption: " + plaintext);

        // Encrypt
        String encrypted = encrypt(plaintext, secretKey);
        System.out.println("After Encryption: " + encrypted);

        // Decrypt
        String decrypted = decrypt(encrypted, secretKey);
        System.out.println("Decrypted message: " + decrypted);
    }
}

import java.util.*;

class VernamCipher {
    public String encode(String ptxt, String key) {
        String ctext = "";
        if (ptxt.length() == key.length()) {
            for (int i = 0; i < ptxt.length(); i++) {
                int p_index = ptxt.charAt(i) - 65;
                int k_index = key.charAt(i) - 65;
                int c_index = p_index + k_index;
                if (c_index >= 26) {
                    c_index = c_index - 26;
                }
                char ctxt = (char) (c_index + 65);
                ctext += ctxt;
            }
        }
        return ctext;
    }

    public String decode(String ctxt, String key) {
        String ptext = "";
        if (ctxt.length() == key.length()) {
            for (int i = 0; i < ctxt.length(); i++) {
                int c_index = ctxt.charAt(i) - 65;
                int k_index = key.charAt(i) - 65;
                int p_index = c_index - k_index;
                if (p_index < 0) {
                    p_index = p_index + 26;
                }
                char ptxt = (char) (p_index + 65);
                ptext += ptxt;
            }
        }




        return ptext;
    }
}

import java.util.*;

class Columnar {
    public void encode(String ptext, int row, int col) {
        Scanner sc = new Scanner(System.in);
        char[][] mat = new char[row][col];
        int index = 0;
        boolean isvalue = true;
        String ctext = "";

        if (ptext.length() <= row * col) {
            // Fill matrix row-wise
            for (int i = 0; i < row; i++) {
                for (int j = 0; j < col; j++) {
                    if (index < ptext.length()) {
                        mat[i][j] = ptext.charAt(index);
                        index++;
                    } else {
                        mat[i][j] = '_'; // padding
                    }
                }
            }

            // Print matrix
            System.out.println("\nMatrix:");
            for (int i = 0; i < row; i++) {
                for (int j = 0; j < col; j++) {
                    System.out.print(mat[i][j] + " ");
                }
                System.out.println();
            }

            // Take key input
            System.out.print("\nEnter keys (permutation of 0-" + (col - 1) + "): ");
            String keys = sc.next();

            if (keys.length() == col) {
                for (int j = 0; j < keys.length(); j++) {
                    int key = Character.getNumericValue(keys.charAt(j));
                    if (key >= col) {
                        isvalue = false;
                        break;
                    }
                }

                if (isvalue) {
                    for (int z = 0; z < keys.length(); z++) {
                        int key = Character.getNumericValue(keys.charAt(z));
                        for (int i = 0; i < row; i++) {
                            ctext += mat[i][key];
                        }
                    }
                } else {
                    System.out.println("Keys out of range!");
                }
            } else {
                System.out.println("Key length must match number of columns!");
            }
        } else {
            System.out.println("Plain text does not fit in matrix!");
        }

        System.out.println("\nCipher Text: " + ctext);
    }
}

    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        c.encode(ptext.toLowerCase(), row, col);
    }

import java.util.*;
import java.math.*;

class DiffieHellman {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        // Input prime number q
        System.out.print("Enter prime number q: ");
        BigInteger q = sc.nextBigInteger();

        // Input primitive root a
        System.out.print("Enter primitive root a: ");
        BigInteger a = sc.nextBigInteger();

        // Input private key of A (XA)
        System.out.print("Enter private key of A (XA): ");
        BigInteger XA = sc.nextBigInteger();

        // Compute public key of A (YA)
        BigInteger YA = a.modPow(XA, q);
        System.out.println("Public key of A (YA): " + YA);

        // Input private key of B (XB)
        System.out.print("Enter private key of B (XB): ");
        BigInteger XB = sc.nextBigInteger();

        // Compute public key of B (YB)
        BigInteger YB = a.modPow(XB, q);
        System.out.println("Public key of B (YB): " + YB);
// Compute secret keys
        BigInteger keyA = YB.modPow(XA, q);
        BigInteger keyB = YA.modPow(XB, q);

        System.out.println("Secret key of A: " + keyA);
        System.out.println("Secret key of B: " + keyB);

        // Verify both keys
        if (keyA.equals(keyB)) {
            System.out.println("Exchange Successful! Shared Secret = " + keyA);
        } else {
            System.out.println("Exchange Failed.");
        }
    }
}
