import hashlib
import time

class Block:
    def __init__(self,index,timestamp,data,prior_hash=0):
        self.index = index 
        self.timestamp = timestamp
        self.data = data
        self.prior_hash = prior_hash
        self.hash = self.create_hash()

    def create_hash(self):
        blockstr = f"{self.index}{self.timestamp}{self.data}{self.prior_hash}".encode()
        return hashlib.sha256(blockstr).hexdigest()

class Blockchain:
    def __init__(self):
        self.chain=[self.create_genesis()]

    def create_genesis(self):
        return Block(0,str(time.localtime()),'',0)

    def get_latest_block(self):
        return self.chain[-1]
    
    def add_new_block(self,newblock):
        newblock.prior_hash = self.get_latest_block().hash
        newblock.hash = newblock.create_hash()
        self.chain.append(newblock)

    def display_blockchain(self):
        for i in range(len(self.chain)):
            print("Block no:",i,"Data:",self.chain[i].data)
            print("Block no:",i,"Hash:",self.chain[i].hash)
            print("Block no:",i,"Prior Hash:",self.chain[i].prior_hash,"\n")

    def checkBC(self):
        for i in range(1,len(self.chain)):
            current = self.chain[i]
            prev = self.chain[i-1]
            if current.hash!=current.create_hash():
                return False
            if current.prior_hash!=prev.hash:
                return False
        return True

// Custom errors
error InsufficientAmount(address sender);
error InsufficientDonationAmount(address sender, uint amt);

contract Donation {

    // Event
    event DonationMoreThan5Ether(address donor, uint amt);

    address owner;

    // Constructor
    constructor() {
        owner = msg.sender;
    }

    // Modifier
    modifier only_owner(){
        require(msg.sender == owner,"You're not the owner");
        _;
    }

    // Donate function
    function donate() public payable {

        if(msg.value == 0){
            revert InsufficientDonationAmount(msg.sender,msg.value);
        }

        if(msg.sender.balance <= 85e18){
            revert InsufficientAmount(msg.sender);
        }

        if(msg.value > 5e18){
            emit DonationMoreThan5Ether(msg.sender,msg.value);
        }
    }

    // Withdraw function
    function withdraw() public payable only_owner returns(bool){

        uint curr_bal = address(this).balance;

        (bool success,) = owner.call{value: curr_bal}("");

        return success;
    }

    // View balance
    function viewMyBalance() public view returns(uint){
        return msg.sender.balance;
    }
}

// SPDX-License-Identifier: MIT
pragma solidity ^0.8.18;

import "@openzeppelin/contracts/token/ERC721/extensions/ERC721URIStorage.sol";
import "@openzeppelin/contracts/utils/Counters.sol";
import "@openzeppelin/contracts/access/Ownable.sol";

contract MyNft is ERC721URIStorage, Ownable {

    using Counters for Counters.Counter;
    Counters.Counter private _tokenIds;

    constructor() ERC721("Python", "PY") Ownable(msg.sender) {}

    function mintNFT(address recipient, string memory tokenURI)
        public
        onlyOwner
        returns(uint256)
    {
        _tokenIds.increment();
        uint256 newItemId = _tokenIds.current();

        _mint(recipient, newItemId);
        _setTokenURI(newItemId, tokenURI);

        return newItemId;
    }
}
