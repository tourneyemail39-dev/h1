pragma solidity ^0.8.0;

contract Donation {

    address public owner;

    constructor() {
        owner = msg.sender;
    }

    // Function to donate ETH
    function donate() public payable {}

    // Owner withdraws all ETH
    function withdraw() public returns (bool) {

        require(msg.sender == owner, "Only owner can withdraw");

        uint256 ctr_bal = address(this).balance;

        (bool success,) = owner.call{value: ctr_bal}("");

        return success;
    }

    // View balance of caller
    function viewMyBalance() public view returns(uint256){
        uint256 myBalance = msg.sender.balance;
        return myBalance;
    }
}
